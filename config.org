*** Custom split
#+BEGIN_SRC emacs-lisp
  (use-package emacs
    :ensure nil
    :preface
    (defun split-and-follow-vertically ()
      "Split window right."
      (interactive)
      (split-window-right)
      (other-window 1))
    :config
    (global-set-key (kbd "C-x 3") #'split-and-follow-vertically))
#+END_SRC
