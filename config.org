  (use-package emacs
    :config
    (setq frame-title-format '("emacs"))

    (tool-bar-mode -1)
    (menu-bar-mode -1)

    ;; Disable startup screen
    (setq inhibit-startup-screen t))

  (use-package emacs
    :ensure nil
    :preface
    (defun split-and-follow-vertically ()
      "Split window right."
      (interactive)
      (split-window-right)
      (other-window 1))
    :config
    (global-set-key (kbd "C-x 3") #'split-and-follow-vertically))
#+END_SRC
